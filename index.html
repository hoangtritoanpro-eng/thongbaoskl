  function loadTemplate() {

    if(templates[val]) {

  async function runGate() {
    const content = document.getElementById('zalo-content');
    if(content.contentEditable === "true") toggleEdit();
    content.innerHTML = "<div style='text-align: center; margin-top: 20px;'><i class='fa-solid fa-circle-notch fa-spin fa-2x' style='color: #0068ff;'></i><br><br><b>Đang kết nối AI...</b></div>";
    
    let danhXung = document.querySelector('input[name="danhxung"]:checked').value;
    const formData = {
      apiKey: document.getElementById('inp-apikey').value, danhXung: danhXung,
      tenLop: document.getElementById('inp-lop').value, suKien: document.getElementById('inp-sukien').value,
      thoiLuong: document.getElementById('inp-thoiluong').value, ngayBatDau: document.getElementById('inp-ngaybatdau').value,
      ngayDiHocLai: document.getElementById('inp-ngaydihoclai').value, yeuCau: document.getElementById('inp-yeucau').value

    try {
      const result = await callBackend('generate', formData);
      document.getElementById('zalo-content').innerText = result.replace(/\*/g, '');
    } catch (e) {
      document.getElementById('zalo-content').innerText = "Lỗi kết nối: " + e;
    }
  }

  function toggleEdit() {
    const content = document.getElementById('zalo-content');
    const btn = document.getElementById('btn-edit');
    if(content.innerText.includes('Đang kết nối AI') || content.innerText.includes('Hệ thống AI đã sẵn sàng')) return;

    if (content.contentEditable === "true") {
      content.contentEditable = "false";
      btn.innerHTML = '<i class="fa-solid fa-pen"></i> Sửa nhanh';
      btn.style.background = "var(--purple-btn)";
    } else {
      content.contentEditable = "true"; content.focus();
      btn.innerHTML = '<i class="fa-solid fa-check"></i> Hoàn tất';
      btn.style.background = "linear-gradient(135deg, #f5af19, #f12711)";
    }
  }

  async function saveToCloud() {
    const content = document.getElementById('zalo-content');
    const text = content.innerText;
    if(!text || text.includes('Đang kết nối') || text.includes('Hệ thống AI đã sẵn sàng')) {
      alert("Chưa có nội dung thông báo để lưu!"); return;
    }
    if(content.contentEditable === "true") toggleEdit();

    const btn = document.getElementById('btn-cloud-save');
    btn.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Đang tải...';
    
    try {
      const response = await callBackend('save', text);
      btn.innerHTML = '<i class="fa-solid fa-cloud-arrow-up"></i> Lưu & Chia sẻ';
      if(response === "SUCCESS") alert("🎉 Đã upload thành công lên Ngân hàng mẫu chung toàn hệ thống!");
      else alert(response);
    } catch(e) {
      alert("Lỗi kết nối: " + e);
      btn.innerHTML = '<i class="fa-solid fa-cloud-arrow-up"></i> Lưu & Chia sẻ';
    }
  }

  async function openHistory() {
    const container = document.getElementById('history-container');
    container.innerHTML = "<div style='text-align: center; padding: 30px;'><i class='fa-solid fa-circle-notch fa-spin fa-2x' style='color: #0068ff;'></i><br><br>Đang đồng bộ dữ liệu đám mây...</div>";
    document.getElementById('historyModal').style.display = 'flex';

    try {
      const history = await callBackend('getHistory', {});
      if(!history || history.length === 0) {
        container.innerHTML = '<div class="no-data"><i class="fa-solid fa-cloud-open fa-3x" style="color:#cbd5e1;"></i><br>Ngân hàng đám mây chưa có dữ liệu mẫu. Hãy là người đầu tiên đóng góp tin nhắn mẫu!</div>';
      } else {
        container.innerHTML = history.map((item, index) => `
          <div class="history-item">
            <div class="history-date"><i class="fa-regular fa-clock"></i> Đóng góp vào lúc: ${item.date}</div>
            <div class="history-text" id="hist-${index}">${item.content}</div>
            <button class="btn-copy-sm" onclick="copyFromHistory('hist-${index}')"><i class="fa-solid fa-copy"></i> Sử dụng mẫu này</button>
          </div>
        `).join('');
      }
    } catch(e) {
      container.innerHTML = "Lỗi tải dữ liệu: " + e;
    }
  }

  function closeHistory() { document.getElementById('historyModal').style.display = 'none'; }
  function copyZalo() {
    const content = document.getElementById('zalo-content');
    if(content.contentEditable === "true") toggleEdit();
    navigator.clipboard.writeText(content.innerText).then(() => alert("✅ Đã copy thông báo vào bộ nhớ đệm!"));
  }
  function copyFromHistory(id) {
    navigator.clipboard.writeText(document.getElementById(id).innerText).then(() => alert("✅ Đã lấy mẫu thành công! Bạn có thể dán thẳng vào nhóm Zalo."));
  }
